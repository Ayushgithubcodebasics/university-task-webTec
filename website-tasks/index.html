15Function: arithmetic operations based on user choice.
Code
function arithmetic(a, b, op){
  switch(op){
    case '+': return a + b;
    case '-': return a - b;
    case '*': return a * b;
    case '/': return b === 0 ? 'Error: divide by zero' : a / b;
  }
}
function t15run(){
  const a = Number(document.getElementById('t15a').value);
  const b = Number(document.getElementById('t15b').value);
  const op = document.getElementById('t15op').value;
  show('t15out', a + ' ' + op + ' ' + b + ' = ' + arithmetic(a,b,op));
}

16Function: calculate factorial.
Code
function factorial(n){
  if(n < 0) return 'Undefined for negatives';
  let r = 1;
  for(let i = 2; i <= n; i++) r *= i;
  return r;
}
function t16run(){
  const n = Number(document.getElementById('t16in').value);
  show('t16out', 'Factorial of ' + n + ' = ' + factorial(n));

21Popup: detect numeric vs string and apply allowed operations.
Code
function t21run(){
  const x = prompt('Enter first value:', '10');
  if(x === null){ show('t21out','Cancelled.'); return; }
  const y = prompt('Enter second value:', '20');
  if(y === null){ show('t21out','Cancelled.'); return; }
  const xNum = x.trim() !== '' && !isNaN(x);
  const yNum = y.trim() !== '' && !isNaN(y);
  let result;
  if(xNum && yNum){
    result = 'Both numeric -> addition: ' + x + ' + ' + y +
             ' = ' + (Number(x) + Number(y));
  } else if(!xNum && !yNum){
    result = 'Both strings -> concatenation: "' + x + '" + "' + y +
             '" = "' + (x + y) + '"';
  } else {
    result = 'Mixed types -> no common operation applied.';
  }
  show('t21out', result);
}

Two popup inputs. If both numeric → add them. If both strings → concatenate. Mixed → reports mixed.

29Countdown timer using setInterval with Start/Pause/Reset.
Code
let t29timer = null, t29count = 30;
function t29render(){ document.getElementById('t29out').textContent = t29count; }
function t29start(){
  if(t29timer) return;                       // already running
  if(t29count <= 0)
    t29count = Number(document.getElementById('t29in').value) || 30;
  t29timer = setInterval(() => {
    t29count--;
    t29render();
    if(t29count <= 0){
      clearInterval(t29timer); t29timer = null;
      document.getElementById('t29out').textContent = "Time's up!";
    }
  }, 1000);
}
function t29pause(){ clearInterval(t29timer); t29timer = null; }  // value kept
function t29reset(){
  clearInterval(t29timer); t29timer = null;
  t29count = Number(document.getElementById('t29in').value) || 30;
  t29render();
}
